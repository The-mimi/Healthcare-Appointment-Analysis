# Define a function to remove outliers based on IQR
def remove_outliers_iqr(data, column):
    Q1 = data[column].quantile(0.25)
    Q3 = data[column].quantile(0.75)
    IQR = Q3 - Q1
    lower_bound = Q1 - 1.5 * IQR
    upper_bound = Q3 + 1.5 * IQR
    return data[(data[column] >= lower_bound) & (data[column] <= upper_bound)]

import numpy as np
import matplotlib.pyplot as plt

# Assuming df is your DataFrame

conditions = ['Alcoholism', 'Hypertension', 'Diabetes', 'Handicap']
showed_up_counts = []
did_not_show_up_counts = []

for condition in conditions:
    showed_up_count = len(df[(df[condition] == True) & (df['Showed_up'] == True)])
    did_not_show_up_count = len(df[(df[condition] == True) & (df['Showed_up'] == False)])

    showed_up_counts.append(showed_up_count)
    did_not_show_up_counts.append(did_not_show_up_count)

# Plotting
x = np.arange(len(conditions))
width = 0.35

fig, ax = plt.subplots(figsize=(10, 6))
rects1 = ax.bar(x - width/2, showed_up_counts, width, label='Showed Up', color='cadetblue')
rects2 = ax.bar(x + width/2, did_not_show_up_counts, width, label='Did Not Show Up', color='firebrick')

ax.set_ylabel('Count')
ax.set_title('Showed Up vs Did Not Show Up by Condition')
ax.set_xticks(x)
ax.set_xticklabels(conditions)
ax.legend()

# Annotate percentages
for i, (showed_up, did_not_show_up) in enumerate(zip(showed_up_counts, did_not_show_up_counts)):
    total = showed_up + did_not_show_up

    # Annotate 'Showed Up'
    if showed_up > 0:
        percentage = f'{100 * showed_up / total:.1f}%'
        x_pos = x[i] - width / 2  # Centering the 'Showed Up' annotation
        y_pos = showed_up
        ax.annotate(percentage, (x_pos, y_pos), ha='center', va='bottom')

    # Annotate 'Did Not Show Up'
    if did_not_show_up > 0:
        percentage = f'{100 * did_not_show_up / total:.1f}%'
        x_pos = x[i] + width/2  # Centering the 'Did Not Show Up' annotation
        y_pos = did_not_show_up
        ax.annotate(percentage, (x_pos, y_pos), ha='center', va='bottom')

plt.show()

# Among all patients, what percentage both had X-condition and showed up

conditions = ['Alcoholism', 'Hypertension', 'Diabetes', 'Handicap']
showed_up_counts = []
did_not_show_up_counts = []

# Loop to calculate counts for each condition
for condition in conditions:
    showed_up_count = len(df[(df[condition] == True) & (df['Showed_up'] == True)])
    did_not_show_up_count = len(df[(df[condition] == True) & (df['Showed_up'] == False)])

    showed_up_counts.append(showed_up_count)
    did_not_show_up_counts.append(did_not_show_up_count)

#  "Showed Up" and "Did Not Show Up"
total_showed_up = len(df[df['Showed_up'] == True])
total_did_not_show_up = len(df[df['Showed_up'] == False])

# Calculate the overall total
overall_total = total_showed_up + total_did_not_show_up

#overall percentage
overall_showed_up_percentage = (total_showed_up / overall_total) * 100 if overall_total > 0 else 0
overall_did_not_show_up_percentage = (total_did_not_show_up / overall_total) * 100 if overall_total > 0 else 0

print(f"Overall Showed Up: {overall_showed_up_percentage:.1f}%")
print(f"Overall Did Not Show Up: {overall_did_not_show_up_percentage:.1f}%")

#  Print percentages for each condition
print("\nPercentages for each condition based on overall total:")
for i, (showed_up, did_not_show_up) in enumerate(zip(showed_up_counts, did_not_show_up_counts)):
    total = showed_up + did_not_show_up

    if total > 0:  # Avoid division by zero
        showed_up_percentage = 100 * showed_up / overall_total
        did_not_show_up_percentage = 100 * did_not_show_up / overall_total
        print(f"{conditions[i]} - Showed Up: {showed_up_percentage:.1f}% | Did Not Show Up: {did_not_show_up_percentage:.1f}%")
    else:
        print(f"{conditions[i]} - No data available (both Showed Up and Did Not Show Up are 0)")

--output-dir=<Unicode>
    Directory to write output(s) to. Defaults
                                  to output to the directory of each notebook. To recover
                                  previous default behaviour (outputting to the current
                                  working directory) use . as the flag value.
    Default: ''
    Equivalent to: [--FilesWriter.build_directory]
--reveal-prefix=<Unicode>
    The URL prefix for reveal.js (version 3.x).
            This defaults to the reveal CDN, but can be any url pointing to a copy
            of reveal.js.
            For speaker notes to work, this must be a relative path to a local
            copy of reveal.js: e.g., "reveal.js".
            If a relative path is given, it must be a subdirectory of the

            Use this to downgrade notebooks.
    Choices: any of [1, 2, 3, 4]
    Default: 4
    Equivalent to: [--NotebookExporter.nbformat_version]

Examples
--------

    The simplest way to use nbconvert is

            > jupyter nbconvert mynotebook.ipynb --to html

            Options include ['asciidoc', 'custom', 'html', 'latex', 'markdown', 'notebook', 'pdf', 'python', 'qtpdf', 'qtpng', 'rst', 'script', 'slides', 'webpdf'].

            > jupyter nbconvert --to latex mynotebook.ipynb

            Both HTML and LaTeX support multiple output templates. LaTeX includes
            'base', 'article' and 'report'.  HTML includes 'basic', 'lab' and
            'classic'. You can specify the flavor of the format used.

            > jupyter nbconvert --to html --template lab mynotebook.ipynb

            You can also pipe the output to stdout, rather than a file

            > jupyter nbconvert mynotebook.ipynb --stdout

            PDF is generated via latex

            > jupyter nbconvert mynotebook.ipynb --to pdf

            You can get (and serve) a Reveal.js-powered slideshow

            > jupyter nbconvert myslides.ipynb --to slides --post serve

            Multiple notebooks can be given at the command line in a couple of
            different ways:

            > jupyter nbconvert notebook*.ipynb
            > jupyter nbconvert notebook1.ipynb notebook2.ipynb

            or you can specify the notebooks list in a config file, containing::

                c.NbConvertApp.notebooks = ["my_notebook.ipynb"]

            > jupyter nbconvert --config mycfg.py

To see all available configurables, use `--help-all`.
